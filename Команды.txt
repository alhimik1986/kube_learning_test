# Создание пода
kubectl create namespace my-namespace
kubectl run hello --image=httpd:latest --port=80 -n my-namespace

# Удаление пода
kubectl delete pods hello -n my-namespace

# Создание пода в default namespace
kubectl run hello --image=httpd:latest --port=80 -n default

# Информация о поде
kubectl describe pods hello

# Запустить команду в поде
kubectl exec hello -- date
kubectl exec -it hello -- sh

# Логи контейнера/пода
kubectl logs hello

# Открываем доступ к поду через веб
kubectl port-forward hello 80:80


# Запустить манифест
kubectl apply -f pod-myweb-lite.yml

# Удалить поды в манифесте (чтобы можно было запустить модифицированный манифест)
kubectl delete -f pod-myweb-lite.yml

# Запустить манифест
kubectl apply -f pod-myweb-ver1.yml
# Снова перенаправляем порты к поду, чтобы сайт пода открывался в браузере
kubectl port-forward my-web 80:80

# Запускаем другой похожий манифест
kubectl delete -f pod-myweb-ver1.yml
kubectl apply -f pod-myweb-ver2.yml
kubectl port-forward my-web 80:80 8080:8080

# Если нода (сервер) упадёт и затем снова поднимется, то все поды внутри него исчезнут и их нужно будет создавать заново.
# Поэтому нужен deployment
# Хотя в minikube такого не произошло (поды не исчезли).


# Список деплойментов
kubectl get deployments
# или 
kubectl get deploy

# Добавить деплоймент
kubectl create deployment my-deployment --image nginx:latest

# Посмотрели созданный деплоймент и увидели ещё один под, созданный деплойментом
kubectl get deploy
kubectl get pods
kubectl describe pods my-deployment-8f9c8c68d-8dvvr
kubectl describe deployment my-deployment

# Создаём реплики подов в деплойменте
kubectl scale deployment my-deployment --replicas 3
# Смотрим появившиеся поды
kubectl get pods
kubectl get deploy
# Смотрип реплика-сеты (набор реплик, кол-во реплик)
kubectl get rs

# Удаляем какой-нибудь из подов
kubectl delete pods my-deployment-8f9c8c68d-8dvvr
# И видим, что вместо него появился новый под
kubectl get pods

# Создаём автоскейл (масштабирование) деплоймента с указанным количеством ресуров
kubectl autoscale deployment my-deployment --min=3 --max=5 --cpu=80
# Проверяем наш деплоймент
kubectl get deploy
# Проверяем созданное масштабирование horizontal pod autoscaler (hpa)
kubectl get hpa
kubectl rollout history deployment/my-deployment
kubectl rollout status deployment/my-deployment

# Ищем название контейнера, чтобы его изменить
kubectl describe deployment my-deployment
# Меняем (обновляем/откатываем) версию контейнера
kubectl set image deployment/my-deployment nginx=nginx:1.28 --record
# Наблюдаем процесс обновления/отката контейнера
kubectl rollout status deployment/my-deployment
# Смотрим историю откатов/обновления деплоймента
kubectl rollout history deployment/my-deployment
# Снова меняем версию контейнера
kubectl set image deployment/my-deployment nginx=nginx:latest --record
# Смотрим историю обновлений/откатов деплоймента
kubectl rollout history deployment/my-deployment

# Откатываем деплоймент на один шаг
kubectl rollout undo deployment/my-deployment
kubectl rollout history deployment/my-deployment
# Откатываем деплоймент на определённую ревизию
kubectl rollout undo deployment/my-deployment --to-revision=3

# Обновить latest-версии контейнеров в деплойменте
kubectl rollout restart deployment/my-deployment
kubectl rollout status deployment/my-deployment

# Создаём новый сервис с типом ClusterIP. Раскрываем (открываем) в деплойменте 80-й порт/
kubectl expose deployment my-deployment --type=ClusterIP --port=80
kubectl get services
# или
kubectl get svc
# Находим IP-адрес сервиса. Заходим в ноду и выполняем команду, в которой указывем найденный IP-адрес
curl 10.108.110.203
# Удаляем созданный сервис
kubectl delete service my-deployment
# Проверяем, что сервис удалён, а деплоймент - нет
kubectl get svc
kubectl get deploy

# Создаём новый сервис с типом NodePort
kubectl expose deployment my-deployment --type=NodePort --port 80
kubectl get services
kubectl describe nodes
kubectl describe nodes | grep ExternalIP
# Здесь мы должны были получить ExternalIP и в kubectl get services получить внешний порт. И по ним должен был открыться сайт.
# Но он не открывается (http://192.168.49.2:31279).

# Удаляем сервис
kubectl delete svc my-deployment


# Создаём новый сервис с типом LoadBalancer
